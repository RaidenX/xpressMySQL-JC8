$ git init 
// Inisiasi folder yang akan di-upload ke GitHub

$ git add . 
// Memasukkan file ke repository folder lokal gitHub yang akan di commit

$ git commit -m "First commit" 
// proses "commit" agar nantinya bisa dipantau setiap perubahan di dalam masing-masing file

$ git remote add origin remote repository URL 
// Menghubungkan antara repositori lokal ke repositori GitHub

$ git remote -v 
// Verifikasi remote URL yang baru dihubungkan

$ git push origin master 
// Mengirim perubahan file dalam repositori lokal ke repositori GitHub

$ git push -f origin master
